\chapter{APPENDIX}

\section{Minimum Software Requirements}
\medskip
\textbf{Operating System}: Windows, macOS, or Linux
\\\\
\textbf{Web Browser}: Modern web browser (Google Chrome, Mozilla Firefox, Microsoft Edge, Safari) with JavaScript enabled.
\\\\
\textbf{Environment (For Local Hosting)}: Node.js (v16 or higher) and npm/yarn.

\section{Minimum Hardware Requirements}
\textbf{Processor}: Any processor capable of running modern web browsers smoothly.
\\\\
\textbf{RAM}: 4 GB (8 GB recommended for optimal multi-tab performance)
\\\\
\textbf{Storage}: 200 MB free space for local codebase and dependencies.
\\\\
\textbf{Display}: A monitor with at least 1280x720 resolution for clear visual data representations.
\\\\
\textbf{Input Devices}: Keyboard and mouse/touchscreen for data input and navigation.
\\\\
\textbf{Network Connectivity}: A stable internet connection to communicate securely with the MERN backend APIs and the MongoDB cloud database.
\\\\
